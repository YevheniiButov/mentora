from flask import request
import re

def is_mobile_request():
    """–û–ø—Ä–µ–¥–µ–ª—è–µ—Ç –º–æ–±–∏–ª—å–Ω—ã–µ —É—Å—Ç—Ä–æ–π—Å—Ç–≤–∞"""
    user_agent = request.headers.get('User-Agent', '').lower()
    mobile_patterns = [
        r'mobile', r'android', r'iphone', r'ipod', 
        r'blackberry', r'windows phone', r'opera mini'
    ]
    return any(re.search(pattern, user_agent) for pattern in mobile_patterns)

def get_device_type():
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Ç–∏–ø —É—Å—Ç—Ä–æ–π—Å—Ç–≤–∞"""
    user_agent = request.headers.get('User-Agent', '').lower()
    if 'tablet' in user_agent or 'ipad' in user_agent:
        return 'tablet'
    elif is_mobile_request():
        return 'mobile'
    return 'desktop'

def measure_performance(f):
    @wraps(f)
